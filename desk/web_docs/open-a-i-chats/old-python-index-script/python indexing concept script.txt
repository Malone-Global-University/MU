import os

def make_index(path):
    files = os.listdir(path)
    with open(os.path.join(path, 'index.html'), 'w') as f:
        f.write("<ul>\n")
        for file in files:
            if file != 'index.html':
                f.write(f'<li><a href="{file}">{file}</a></li>\n')
        f.write("</ul>")

for root, dirs, _ in os.walk('.'):
    make_index(root)
